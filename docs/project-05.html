void RenderControlPanel() {
    ImGui::Begin("жҺ§еҲ¶йқўжқҝ");
    
    // жҺЁжөҒжҺ§еҲ¶
    if (streamer_.GetState() == StreamerState::IDLE) {
        if (ImGui::Button("ејҖе§ӢжҺЁжөҒ", ImVec2(120, 40))) {
            streamer_.Start();
        }
    } else if (streamer_.GetState() == StreamerState::STREAMING) {
        if (ImGui::Button("еҒңжӯўжҺЁжөҒ", ImVec2(120, 40))) {
            streamer_.Stop();
        }
    }
    
    ImGui::Separator();
    
    // зҫҺйўңеҸӮж•°
    auto params = streamer_.GetBeautyParams();
    if (ImGui::SliderFloat("зЈЁзҡ®", &params.smooth, 0.0f, 1.0f)) {
        streamer_.SetBeautyParams(params);
    }
    if (ImGui::SliderFloat("зҫҺзҷҪ", &params.whitening, 0.0f, 1.0f)) {
        streamer_.SetBeautyParams(params);
    }
    
    ImGui::Separator();
    
    // и®ҫеӨҮйҖүжӢ©
    if (ImGui::BeginCombo("ж‘„еғҸеӨҙ", current_camera_.c_str())) {
        for (const auto& cam : camera_list_) {
            if (ImGui::Selectable(cam.name.c_str())) {
                streamer_.SwitchCamera(cam.id);
                current_camera_ = cam.name;
            }
        }
        ImGui::EndCombo();
    }
    
    ImGui::End();
}

class StatsPanel {
public:
    void Update(const StreamStats& stats) {
        // ж·»еҠ еҲ°еҺҶеҸІ
        video_bitrate_history_.push_back(stats.video_bitrate / 1000.0f);  // kbps
        if (video_bitrate_history_.size() > 100) {
            video_bitrate_history_.erase(video_bitrate_history_.begin());
        }
    }
    
    void Render() {
        ImGui::Begin("жҺЁжөҒз»ҹи®Ў");
        
        // з ҒзҺҮжӣІзәҝ
        ImGui::PlotLines("и§Ҷйў‘з ҒзҺҮ (kbps)",
                         video_bitrate_history_.data(),
                         video_bitrate_history_.size(),
                         0, nullptr, 0.0f, 5000.0f,
                         ImVec2(0, 100));
        
        // ж•°еӯ—жҳҫзӨә
        ImGui::Text("и§Ҷйў‘: %.0f kbps @ %.1f fps",
                    current_stats_.video_bitrate / 1000.0f,
                    current_stats_.video_fps);
        ImGui::Text("йҹійў‘: %.0f kbps",
                    current_stats_.audio_bitrate / 1000.0f);
        ImGui::Text("е·ІжҺЁжөҒ: %02d:%02d:%02d",
                    hours_, minutes_, seconds_);
        
        // иҝһжҺҘзҠ¶жҖҒ
        const char* status_str[] = {"з©әй—І", "иҝһжҺҘдёӯ", "жҺЁжөҒдёӯ", "й”ҷиҜҜ"};
        ImVec4 status_color[] = {
            ImVec4(0.5f, 0.5f, 0.5f, 1.0f),  // зҒ°
            ImVec4(1.0f, 1.0f, 0.0f, 1.0f),  // й»„
            ImVec4(0.0f, 1.0f, 0.0f, 1.0f),  // з»ҝ
            ImVec4(1.0f, 0.0f, 0.0f, 1.0f)   // зәў
        };
        ImGui::TextColored(status_color[status_], "зҠ¶жҖҒ: %s", status_str[status_]);
        
        ImGui::End();
    }
    
private:
    std::vector<float> video_bitrate_history_;
    StreamStats current_stats_;
    int hours_ = 0, minutes_ = 0, seconds_ = 0;
    int status_ = 0;
};

void RenderVolumeMeter(float db) {
    // е°Ҷ dB жҳ е°„еҲ° 0-1
    float level = (db + 60.0f) / 60.0f;  // -60dB ~ 0dB
    level = std::clamp(level, 0.0f, 1.0f);
    
    ImDrawList* draw_list = ImGui::GetWindowDrawList();
    ImVec2 pos = ImGui::GetCursorScreenPos();
    ImVec2 size(200, 20);
    
    // иғҢжҷҜ
    draw_list->AddRectFilled(pos, ImVec2(pos.x + size.x, pos.y + size.y),
                              IM_COL32(64, 64, 64, 255));
    
    // йҹійҮҸжқЎ
    ImU32 color;
    if (level < 0.7f) {
        color = IM_COL32(0, 255, 0, 255);    // з»ҝ
    } else if (level < 0.9f) {
        color = IM_COL32(255, 255, 0, 255);  // й»„
    } else {
        color = IM_COL32(255, 0, 0, 255);    // зәў
    }
    
    draw_list->AddRectFilled(pos,
                              ImVec2(pos.x + size.x * level, pos.y + size.y),
                              color);
    
    ImGui::Dummy(size);  // еҚ дҪҚ
}
